    c = (a + b)/2
    count = 0

    while(abs(f(c)) > 10**(-7)):
        if(f(c)*f(b) < 0):
            a = c
        else:
            b = c
        count += 1
        c = (a + b)/2
    print("Root in the given interval is:", c)
    print("Number of iteration:", count)

def regulaFalsi(a, b):
    
    c = (a*f(b) - b*f(a))/(f(b) - f(a))
    count = 0
    
    while(abs(f(c)) > 10**(-7)):
        
        if(f(b)*f(c) < 0):
            a = c
        else:
            b = c
        count += 1
        c = (a*f(b) - b*f(a))/(f(b) - f(a))
    print("Root in the given interval is:", c)
    print("Number of iteration is:", count)
